  <script>

    let currentPage = 'homePage';

    const pageHistory = ['homePage'];


    // 全局数据（本地存储）

    const userData = {

      checkinData: JSON.parse(localStorage.getItem('solaCheckin')) || {

        checkedDates: [],

        continuousDays: 0,

        totalCheckin: 0

      },

      merit: parseInt(localStorage.getItem('solaMerit')) || 0

    };


    // 页面切换

    function goPage(id) {

      document.getElementById(currentPage).classList.remove('active');

      document.getElementById(id).classList.add('active');

      if (currentPage !== id) pageHistory.push(id);

      currentPage = id;


      if (id === 'checkinPage') renderCalendar();

      if (id === 'profilePage') updateProfileUI();

    }


    function back() {

      if (pageHistory.length > 1) {

        pageHistory.pop();

        const prev = pageHistory[pageHistory.length - 1];

        goPage(prev);

      }

    }


    function logout() {

      if (confirm('确定要退出登录吗？')) {

        alert('已退出登录');

        goPage('homePage');

      }

    }


    // 每日打卡逻辑

    function renderCalendar() {

      const today = new Date();

      const year = today.getFullYear();

      const month = today.getMonth();

      const firstDay = new Date(year, month, 1);

      const lastDay = new Date(year, month + 1, 0);

      const daysInMonth = lastDay.getDate();

      const startWeekDay = firstDay.getDay();


      document.getElementById('checkinMonth').textContent = `${year}年${String(month+1).padStart(2,'0')}月`;

      const grid = document.getElementById('calendarGrid');

      grid.innerHTML = '';


      // 上月填充

      for (let i=0; i<startWeekDay; i++) {

        const dayEl = document.createElement('div');

        dayEl.className = 'calendar-day other-month';

        grid.appendChild(dayEl);

      }


      // 当月日期

      for (let day=1; day<=daysInMonth; day++) {

        const dateStr = `${year}-${String(month+1).padStart(2,'0')}-${String(day).padStart(2,'0')}`;

        const isToday = day === today.getDate();

        const isChecked = userData.checkinData.checkedDates.includes(dateStr);


        const dayEl = document.createElement('div');

        dayEl.className = 'calendar-day';

        dayEl.textContent = day;

        if (isToday) dayEl.classList.add('today');

        if (isChecked) dayEl.classList.add('checked');

        grid.appendChild(dayEl);

      }


      updateCheckinStats();

      const todayStr = `${year}-${String(month+1).padStart(2,'0')}-${String(today.getDate()).padStart(2,'0')}`;

      document.getElementById('checkinBtn').disabled = userData.checkinData.checkedDates.includes(todayStr);

    }


    function updateCheckinStats() {

      const today = new Date();

      const year = today.getFullYear();

      const month = today.getMonth();

      const monthStr = `${year}-${String(month+1).padStart(2,'0')}`;


      const monthCount = userData.checkinData.checkedDates.filter(d => d.startsWith(monthStr)).length;

      document.getElementById('monthCheckin').textContent = monthCount;

      document.getElementById('currentContinuous').textContent = userData.checkinData.continuousDays;

      document.getElementById('totalCheckinStat').textContent = userData.checkinData.totalCheckin;


      updateProfileUI();

    }


    function doCheckin() {

      const today = new Date();

      const dateStr = `${today.getFullYear()}-${String(today.getMonth()+1).padStart(2,'0')}-${String(today.getDate()).padStart(2,'0')}`;

      if (userData.checkinData.checkedDates.includes(dateStr)) return;


      userData.checkinData.checkedDates.push(dateStr);

      userData.checkinData.totalCheckin += 1;

      userData.checkinData.continuousDays += 1;

      userData.merit += 10;


      localStorage.setItem('solaCheckin', JSON.stringify(userData.checkinData));
