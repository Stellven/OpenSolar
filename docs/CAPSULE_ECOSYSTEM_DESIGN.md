### 1.2 开放标准定义

#### 1.2.1 Capsule URI 标准

```
capsule://<namespace>/<name>@<version>[?<params>]

Examples:
  capsule://solar.core/weather-query@1.0.0
  capsule://community.utils/json-parser@2.1.0?locale=zh-CN
  capsule://org.acme/internal-tool@latest
```

**URI 组成:**

| 部分 | 说明 | 示例 |
|------|------|------|
| `namespace` | 组织/作用域 | `solar.core`, `community.utils` |
| `name` | 胶囊名称 | `weather-query`, `json-parser` |
| `version` | SemVer 版本 | `1.0.0`, `^1.0`, `latest` |
| `params` | 可选参数 | `locale=zh-CN`, `timeout=5000` |

#### 1.2.2 调用协议 (JSON-RPC 2.0 扩展)

```json
// 请求
{
```

#### 1.3.1 联邦注册协议

```typescript
interface FederationRegistry {
  // 注册胶囊到联邦
  register(capsule: CapsuleSpec, provider: AgentID): Promise<RegistrationResult>;

  // 发现可用胶囊
  discover(query: DiscoveryQuery): Promise<CapsuleInfo[]>;

  // 订阅胶囊变更
  subscribe(pattern: string, callback: (event: CapsuleEvent) => void): Subscription;

  // 验证胶囊来源
  verify(capsuleId: CapsuleURI, signature: Signature): Promise<VerificationResult>;
}

interface DiscoveryQuery {
  namespace?: string;          // 命名空间过滤
  capabilities?: string[];     // 能力要求 (如 "network.read", "fs.write")
  min_scare_score?: number;    // 最低 SCARE 分数
  providers?: AgentID[];       // 指定提供者
  exclude_providers?: AgentID[]; // 排除提供者
}
```

#### 1.3.3 信任与声誉系统

```yaml
# 信任评估维度
trust_dimensions:
  # 1. 静态信任 (规约分析)
  static_trust:
    scare_score: 0-100          # SCARE 评分
    spec_completeness: 0-100    # 规约完整度
    test_coverage: 0-100        # 测试覆盖率

  # 2. 动态信任 (执行历史)
  dynamic_trust:
    success_rate: 0-100         # 成功率
    avg_latency_ratio: 0-10     # 延迟/承诺比
    cost_accuracy: 0-100        # 成本预估准确度

  # 3. 社区信任 (声誉)
  community_trust:
    usage_count: number         # 使用次数
    positive_reviews: number    # 正面评价
    incident_count: number      # 事故次数

# 信任分数计算
trust_score =
  static_trust * 0.3 +
  dynamic_trust * 0.5 +
  community_trust * 0.2
```

### 2.2 Token 优化策略

这是关键问题：**如何最小化 Token 消耗？**

```
┌─────────────────────────────────────────────────────────────────────┐
│                   TOKEN OPTIMIZATION STRATEGIES                     │
├─────────────────────────────────────────────────────────────────────┤
│                                                                     │
│  Strategy 1: Lazy Specification Loading (规约懒加载)                │
│  ─────────────────────────────────────────────────────────────────  │
│                                                                     │
│  传统方式 (高 Token):                                               │
│  ┌─────────────────────────────────────────────────────────────┐   │
│  │  加载完整规约到 Context (每个 ~500 tokens)                   │   │
│  │  10 个胶囊 = 5000 tokens 浪费在未使用的规约上                │   │
│  └─────────────────────────────────────────────────────────────┘   │
│                                                                     │
│  优化方式 (低 Token):                                               │
│  ┌─────────────────────────────────────────────────────────────┐   │
│  │  Level 0: 只加载索引 (每个 ~20 tokens)                       │   │
│  │    {id: "weather-query", keywords: ["天气", "weather"]}     │   │
│  │                                                              │   │
│  │  Level 1: 匹配后加载接口 (每个 ~100 tokens)                  │   │
│  │    {input: {...}, output: {...}, errors: [...]}             │   │
│  │                                                              │   │
│  │  Level 2: 执行前加载完整 (~500 tokens, 仅当需要时)           │   │
│  │    完整规约                                                  │   │
│  └─────────────────────────────────────────────────────────────┘   │
│                                                                     │
│  节省: 10个胶囊从 5000 tokens → ~300 tokens (94% 节省)              │
│                                                                     │
├─────────────────────────────────────────────────────────────────────┤
│                                                                     │
│  Strategy 2: Capsule Index Compression (索引压缩)                   │
│  ─────────────────────────────────────────────────────────────────  │
│                                                                     │
│  压缩前 (JSON):                                                     │
│  ┌─────────────────────────────────────────────────────────────┐   │
│  │  {"capsule_id": "capsule://solar.core/weather-query@1.0.0", │   │
│  │   "name": "weather-query", "namespace": "solar.core", ...}  │   │
│  │  ~150 chars = ~40 tokens                                    │   │
│  └─────────────────────────────────────────────────────────────┘   │
│                                                                     │
│  压缩后 (Compact):                                                  │
│  ┌─────────────────────────────────────────────────────────────┐   │
│  │  "C1:weather-query:天气,weather:Q"                          │   │
│  │  ~35 chars = ~10 tokens                                     │   │
│  └─────────────────────────────────────────────────────────────┘   │
│                                                                     │
│  索引编码格式:                                                      │
│  <ID>:<name>:<keywords>:<type>                                     │
│  • ID: 短ID (C1, C2, ...)                                          │
│  • type: Q=Query, A=Action, M=Monitor, T=Transform                 │
│                                                                     │
├─────────────────────────────────────────────────────────────────────┤
│                                                                     │
│  Strategy 3: Result Caching (结果缓存)                              │
│  ─────────────────────────────────────────────────────────────────  │
│                                                                     │
│  缓存层次:                                                          │
│  ┌─────────────────────────────────────────────────────────────┐   │
│  │  L1: In-Memory (会话内)                                      │   │
│  │      TTL: 会话结束                                           │   │
│  │      命中: 0 tokens, 0 ms                                    │   │
│  │                                                              │   │
│  │  L2: SQLite (本地持久)                                       │   │
│  │      TTL: 可配置 (默认 1 小时)                               │   │
│  │      命中: 0 tokens, ~1 ms                                   │   │
│  │                                                              │   │
│  │  L3: Federation (联邦共享)                                   │   │
│  │      TTL: 可配置                                             │   │
│  │      命中: 0 tokens, ~50 ms                                  │   │
│  └─────────────────────────────────────────────────────────────┘   │
│                                                                     │
│  缓存键: hash(capsule_id + input + version)                        │
│  幂等胶囊自动缓存，非幂等胶囊不缓存                                 │
│                                                                     │
├─────────────────────────────────────────────────────────────────────┤
│                                                                     │
│  Strategy 4: Batch Invocation (批量调用)                            │
│  ─────────────────────────────────────────────────────────────────  │
│                                                                     │
│  单独调用 (高 Token):                                               │
│  ┌─────────────────────────────────────────────────────────────┐   │
│  │  invoke(weather, {location: "北京"})  // ~50 tokens         │   │
│  │  invoke(weather, {location: "上海"})  // ~50 tokens         │   │
│  │  invoke(weather, {location: "深圳"})  // ~50 tokens         │   │
│  │  Total: ~150 tokens, 3 次往返                                │   │
│  └─────────────────────────────────────────────────────────────┘   │
│                                                                     │
│  批量调用 (低 Token):                                               │
│  ┌─────────────────────────────────────────────────────────────┐   │
│  │  batch([                                                     │   │
│  │    {capsule: "weather", input: {location: "北京"}},         │   │
│  │    {capsule: "weather", input: {location: "上海"}},         │   │
│  │    {capsule: "weather", input: {location: "深圳"}}          │   │
│  │  ])                                                          │   │
│  │  Total: ~80 tokens, 1 次往返                                 │   │
│  └─────────────────────────────────────────────────────────────┘   │
│                                                                     │
├─────────────────────────────────────────────────────────────────────┤
│                                                                     │
│  Strategy 5: Spec Dehydration (规约脱水)                            │
│  ─────────────────────────────────────────────────────────────────  │
│                                                                     │
│  完整规约 (执行前):                                                 │
│  ┌─────────────────────────────────────────────────────────────┐   │
│  │  apiVersion: capsule/v1                                      │   │
│  │  kind: Capsule                                               │   │
```

### 3.1 组合类型语义

```
┌─────────────────────────────────────────────────────────────────────┐
│                    COMPOSITION SEMANTICS                            │
├─────────────────────────────────────────────────────────────────────┤
│                                                                     │
│  Type 1: Sequence (顺序) - 管道模式                                 │
│  ─────────────────────────────────────────────────────────────────  │
│                                                                     │
│  语法: A >> B >> C                                                  │
│  语义: A 的输出成为 B 的输入，B 的输出成为 C 的输入                 │
│                                                                     │
│  ┌─────┐      ┌─────┐      ┌─────┐                                 │
│  │  A  │ ──▶ │  B  │ ──▶ │  C  │                                 │
│  └─────┘      └─────┘      └─────┘                                 │
│                                                                     │
│  规约派生:                                                          │
│  • duration_max = A.duration + B.duration + C.duration              │
│  • permissions = A.perms ∪ B.perms ∪ C.perms                       │
│  • idempotent = A.idem ∧ B.idem ∧ C.idem                           │
│  • input_schema = A.input_schema                                    │
│  • output_schema = C.output_schema                                  │
│                                                                     │
│  示例:                                                              │
│  fetch_data >> transform_json >> save_to_db                         │
│                                                                     │
├─────────────────────────────────────────────────────────────────────┤
│                                                                     │
│  Type 2: Parallel (并行) - 扇出模式                                 │
│  ─────────────────────────────────────────────────────────────────  │
│                                                                     │
│  语法: [A, B, C]                                                    │
│  语义: A, B, C 同时执行，收集所有结果                               │
│                                                                     │
│          ┌─────┐                                                    │
│       ┌──│  A  │──┐                                                │
│       │  └─────┘  │                                                │
│  ──▶ ─┼─ ┌─────┐ ─┼──▶ [A.out, B.out, C.out]                       │
│       │  │  B  │  │                                                │
│       │  └─────┘  │                                                │
│       │  ┌─────┐  │                                                │
│       └──│  C  │──┘                                                │
│          └─────┘                                                    │
│                                                                     │
│  规约派生:                                                          │
│  • duration_max = max(A.duration, B.duration, C.duration)           │

│                                                                     │

│  ─────────────────────────────────────────────────────────────────  │
│                                                                     │

│                                                                     │

│                                                                     │
│  规约派生:                                                          │
│  • duration_max = max(A.duration, B.duration) // 最坏情况           │
│  • permissions = A.perms ∪ B.perms // 都需要声明                   │
│                                                                     │
│  示例:                                                              │
│  if(has_cache) return_cached else fetch_fresh                       │
│                                                                     │
├─────────────────────────────────────────────────────────────────────┤
│                                                                     │
│  Type 4: Loop (循环) - 迭代模式                                     │
│  ─────────────────────────────────────────────────────────────────  │
│                                                                     │
│  语法: while(cond, max=N) A                                         │
│  语义: 重复执行 A 直到条件不满足或达到最大次数                      │
│                                                                     │
│        ┌──────────────────┐                                        │
│        │                  │                                        │
│        ▼                  │                                        │
│  ──▶ cond? ──▶ ┌─────┐ ──┘                                        │
│        │       │  A  │                                             │

│                                                                     │
```

### 3.4 组合规约派生算法

```typescript
// 组合规约派生 - 从子胶囊自动计算组合胶囊的规约

interface DerivedSpec {
  timing: {
    max_duration: number;
    typical_duration: number;
  };
  permissions: {
    resources: Resource[];
    denies: string[];
  };
  risks: {
    max_cost: number;
    typical_cost: number;
  };
  idempotency: boolean;
}

function deriveSpec(composition: Composition): DerivedSpec {
  const subSpecs = composition.steps.map(s => getSpec(s.capsule_id));

  switch (composition.type) {
    case 'sequence':
      return {
        timing: {
          max_duration: sum(subSpecs, s => s.timing.max_duration),
          typical_duration: sum(subSpecs, s => s.timing.typical_duration),
        },
        permissions: {
          resources: union(subSpecs, s => s.permissions.resources),
          denies: intersection(subSpecs, s => s.permissions.denies),
        },
        risks: {
          max_cost: sum(subSpecs, s => s.risks.max_cost),
          typical_cost: sum(subSpecs, s => s.risks.typical_cost),
        },
        idempotency: all(subSpecs, s => s.idempotency),
      };

    case 'parallel':
      return {
        timing: {
          max_duration: max(subSpecs, s => s.timing.max_duration),
          typical_duration: max(subSpecs, s => s.timing.typical_duration),
        },
        permissions: {
          resources: union(subSpecs, s => s.permissions.resources),
          denies: intersection(subSpecs, s => s.permissions.denies),
        },
        risks: {
          max_cost: sum(subSpecs, s => s.risks.max_cost), // 所有都执行
          typical_cost: sum(subSpecs, s => s.risks.typical_cost),
        },
        idempotency: all(subSpecs, s => s.idempotency),
      };

    case 'conditional':
      return {
        timing: {
          max_duration: max(subSpecs, s => s.timing.max_duration),
          typical_duration: avg(subSpecs, s => s.timing.typical_duration),
        },
        permissions: {
          resources: union(subSpecs, s => s.permissions.resources), // 都可能执行
          denies: intersection(subSpecs, s => s.permissions.denies),
        },
        risks: {
          max_cost: max(subSpecs, s => s.risks.max_cost),
          typical_cost: avg(subSpecs, s => s.risks.typical_cost),
        },
        idempotency: all(subSpecs, s => s.idempotency),
      };

    case 'loop':
      const maxIter = composition.max_iterations || 10;
      const baseSpec = deriveSpec({ type: 'sequence', steps: composition.steps });
      return {
        timing: {
          max_duration: baseSpec.timing.max_duration * maxIter,
          typical_duration: baseSpec.timing.typical_duration * (maxIter / 2),
        },
        permissions: baseSpec.permissions,
        risks: {
          max_cost: baseSpec.risks.max_cost * maxIter,
          typical_cost: baseSpec.risks.typical_cost * (maxIter / 2),
        },
        idempotency: false, // 循环通常非幂等
      };

    case 'fallback':
      return {
        timing: {
          max_duration: sum(subSpecs, s => s.timing.max_duration), // 最坏全试
          typical_duration: subSpecs[0]?.timing.typical_duration || 0, // 典型首选成功
        },
        permissions: {
          resources: union(subSpecs, s => s.permissions.resources),
          denies: intersection(subSpecs, s => s.permissions.denies),
        },
        risks: {
          max_cost: sum(subSpecs, s => s.risks.max_cost),
          typical_cost: subSpecs[0]?.risks.typical_cost || 0,
        },
        idempotency: all(subSpecs, s => s.idempotency),
      };
  }
}
```

```yaml
          capsule_id: capsule://solar.office/email-unread@1.0.0
          input_mapping:
            limit: 5

    # Step 2: 生成报告
    - id: generate_report
      capsule_id: capsule://solar.transform/daily-report-formatter@1.0.0
      input_mapping:
        weather: "collect_data.$.weather"

        tasks: "collect_data.$.tasks"
        emails: "collect_data.$.emails"

    # Step 3: 保存和发送 (带降级)
    - id: deliver
      type: parallel
      steps:
        - id: save_note
          capsule_id: capsule://solar.office/note-create@1.0.0
          input_mapping:
            title: "'日报 - ' + new Date().toLocaleDateString()"
            body: "generate_report.$.report"

        - id: send_email
          type: fallback
          steps:
            - capsule_id: capsule://solar.office/email-send@1.0.0
              input_mapping:
                to: "$.config.email"
                subject: "'每日报告'"
                body: "generate_report.$.report"
            - capsule_id: capsule://solar.office/push-notification@1.0.0
              input_mapping:
                message: "'日报已生成，邮件发送失败'"
```
