brief_has_hold_marker() {
  local sid="$1"
  local brief="$SPRINTS_DIR/${sid}.product-brief.md"
  [[ -f "$brief" ]] || return 1   # 没 brief = 不 hold
  grep -qE '待[^[:space:]]*手动 active|待监护人手动 active|\[DRAFTING-HOLD-NO-AUTODISPATCH\]|hold drafting|禁止再 dispatch' "$brief"
}

#### 2.4 改 `mark_drafting_retry()` (line 2248-2257) 加计数+熔断

新版本 (覆盖原函数):

```bash
mark_drafting_retry() {
  local sid="$1" stage="$2" reason="$3"
  local marker="$HARNESS_DIR/.drafting-flow-retry"
  local count_dir="$HARNESS_DIR/.drafting-failure-counts"
  local count_file="$count_dir/${sid}.${stage}.count"
  local now ts cur_count cur_ts cooldown threshold
  cooldown="${DRAFTING_RETRY_COOLDOWN_SEC:-900}"
  threshold="${DRAFTING_AUTO_HOLD_THRESHOLD:-2}"
  now=$(date +%s)
  ts=$(date -u +"%Y-%m-%dT%H:%MZ")
  mkdir -p "$count_dir"

  # 读取/更新失败计数 (24h 窗口)
  cur_count=0
  cur_ts=0
  if [[ -f "$count_file" ]]; then
    cur_ts=$(cut -d: -f1 < "$count_file")
    cur_count=$(cut -d: -f2 < "$count_file")
    if (( now - cur_ts > 86400 )); then
      cur_count=0   # 24h 重置
    fi
  fi
  cur_count=$((cur_count + 1))
  echo "${now}:${cur_count}" > "$count_file"

  # 阈值熔断 → 写 status.json auto_held=true, 跳过 inbox 写入
  if (( cur_count >= threshold )); then
    local sf="$SPRINTS_DIR/${sid}.status.json"
    if [[ -f "$sf" ]]; then
      python3 -c "
import json, datetime
sf = '$sf'
d = json.load(open(sf))
if not d.get('auto_held'):
    d['auto_held'] = True
    d['auto_held_reason'] = '${cur_count} consecutive ${stage} dispatch failures within 24h (last reason: ${reason})'
    d['auto_held_at'] = datetime.datetime.now(datetime.UTC).strftime('%Y-%m-%dT%H:%M:%SZ')
    d['auto_held_failure_count'] = ${cur_count}
    d['updated_at'] = d['auto_held_at']
    d.setdefault('history', []).append({
        'ts': d['auto_held_at'],
        'event': 'auto_held_by_coordinator',
        'by': 'coordinator',
        'stage': '${stage}',
        'reason': '${reason}',
        'count': ${cur_count}
    })
    json.dump(d, open(sf, 'w'), indent=2)
"
      log "${R}[mark_drafting_retry] AUTO-HELD ${sid} (${cur_count} fails); skipping inbox cooldown spam${N}"
    fi
    # 老的 marker 仍然写 (drafting_retry_blocked 用), 但不写 inbox
    echo "${sid}:${stage}:${now}" >> "$marker"
    return 0
  fi

  # 未达阈值: 走旧逻辑 (写 marker + inbox cooldown)
  echo "${sid}:${stage}:${now}" >> "$marker"
  printf '%s\n' "- [ ] [${ts}] [DRAFTING-DISPATCH-COOLDOWN] ${sid} ${stage} dispatch failed (${reason}); cooldown ${cooldown}s, no pane spam" \
    >> "$PLANNER_INBOX"
}
```

**关键点**:
- 第 1 次失败仍写 inbox (保留可观测性)
- 第 2 次失败立刻 auto_held + 不写 inbox (止血)
- 24h 窗口防"几天前失败累加"误判
- 环境变量 `DRAFTING_AUTO_HOLD_THRESHOLD` 可调
