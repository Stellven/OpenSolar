```bash
~/.solar/bin/solar-verify --red-flags "$sid" >> "$VERIFY_OUTPUT" 2>&1
VERIFY_EXIT_2=$?

# 加时间戳
echo "" >> "$VERIFY_OUTPUT"
echo "## 时间戳" >> "$VERIFY_OUTPUT"
echo "" >> "$VERIFY_OUTPUT"
echo "Generated: $(date -u +"%Y-%m-%dT%H:%M:%SZ")" >> "$VERIFY_OUTPUT"

# 失败不阻塞，但记录
if [ $VERIFY_EXIT_1 -ne 0 ] || [ $VERIFY_EXIT_2 -ne 0 ]; then
  log "${Y}solar-verify failed for $sid (exit $VERIFY_EXIT_1/$VERIFY_EXIT_2), see $VERIFY_OUTPUT${N}"
  echo "**verify-failed**" >> "$VERIFY_OUTPUT"
else
  log "${G}solar-verify passed for $sid, result: $VERIFY_OUTPUT${N}"
fi

dispatch_to_pane "$PANE_EVALUATOR" "" "$sid"
```
