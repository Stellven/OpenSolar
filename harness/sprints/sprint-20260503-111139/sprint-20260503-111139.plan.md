### ingest_codex_all_files

```bash
ingest_codex_all_files() {
  local n_contracts=0 n_reviews=0 n_research=0 n_unknown=0 n_templates=0
  for cf in $(find "$CODEX_INBOX" -maxdepth 1 -name "*.md" ! -name "*.template.md" 2>/dev/null); do
    [ -f "$cf" ] || continue
    local base
    base=$(basename "$cf")
    # dedup
    [ -f "$CODEX_PROCESSED/$base" ] && continue
    case "$base" in
      contract-*|execution-contract-*)
        # 走现有起 sprint 路径
        ingest_single_contract "$cf"
        n_contracts=$((n_contracts + 1))
        ;;
      review-*)
        notify_planner_codex_file "CODEX-REVIEW" "$base"
        cp "$cf" "$CODEX_PROCESSED/$base"
        rm "$cf"
        n_reviews=$((n_reviews + 1))
        ;;
      research-*)
        notify_planner_codex_file "CODEX-RESEARCH" "$base"
        cp "$cf" "$CODEX_PROCESSED/$base"
        rm "$cf"
        n_research=$((n_research + 1))
        ;;
      *)
        notify_planner_codex_file "CODEX-UNKNOWN" "$base"
        cp "$cf" "$CODEX_PROCESSED/$base"
        rm "$cf"
        n_unknown=$((n_unknown + 1))
        ;;
    esac
  done
  echo "[$(date '+%H:%M:%S')] scan: contracts=$n_contracts reviews=$n_reviews research=$n_research unknown=$n_unknown templates_skipped=$n_templates"
}
```

### notify_planner_codex_file

```bash
notify_planner_codex_file() {
  local type="$1" base="$2"
  local ts
  ts=$(date -u +"%Y-%m-%dT%H:%MZ")
  local inbox="$HOME/.solar/harness/PLANNER-INBOX.md"
  mkdir -p "$(dirname "$inbox")"
  printf '%s\n' "- [ ] [${ts}] [${type}] ${base} (~/.solar/codex-bridge/from-codex/)" >> "$inbox"
}
```

### flock 防多开 (D4)

```bash
PID_FILE="$HOME/.solar/harness/.chain-watcher.pid"
exec 200>"$PID_FILE"
flock -n 200 || { echo "[chain-watcher] already running, exit"; exit 0; }
echo $$ > "$PID_FILE"
```

### ingest_codex_contracts 改造

现有函数改为 `ingest_single_contract <filepath>` — 接受单文件路径（不再自带 find），被 `ingest_codex_all_files` 逐文件调用。dedup 和 cp+rm 由调用方（`ingest_codex_all_files`）统一处理。

## 风险点

1. **find 遍历性能**: from-codex/ 目录文件数极少（<50），无性能问题
2. **PLANNER-INBOX.md 不存在**: `mkdir -p` + `>>` 追加自动创建
3. **并发写入 PLANNER-INBOX**: watchdog 熔断通知和 chain-watcher 可能同时写，但 `>>` 在文件系统层面是 append-only atomic（小 write），不会交叉损坏
4. **现有 ingest_codex_contracts 重构**: 需拆分为单文件版本，确保 contract- 路径的 dedup+cp+rm 不遗漏
5. **D6/D7 测试需要 chain-watcher 运行**: 用 timeout + source 函数方式测试，不依赖常驻进程
