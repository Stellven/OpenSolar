# 清空指定 sprint 占用的所有 pane (终态时调用)
# 遍历 PANE_CURRENT_SPRINT, 找匹配 sid 的 idx 并 unset, 然后持久化
clear_pane_assignment() {
  local sid="$1"
  local cleared=0
  for idx in "${!PANE_CURRENT_SPRINT[@]}"; do
    if [[ "${PANE_CURRENT_SPRINT[$idx]}" == "$sid" ]]; then
      unset 'PANE_CURRENT_SPRINT[$idx]'

  if (( cleared > 0 )); then
    save_pane_assignments
  fi
}

### D2 — dispatch_to_pane busy 守卫

**位置**: `coordinator.sh:370` (函数入口), 紧跟 `PANE_PLANNER` 特殊路径之后, `is_pane_present` 检查之前.

```bash
dispatch_to_pane() {
  local pane="$1" message="$2" sid="${3:-dispatch}"

  # ── 规划者特殊路径 (现有, 不动) ──
  if [[ "$pane" == "$PANE_PLANNER" ]] || [[ "$pane" == "$SESSION_NAME:0.0" ]]; then
    notify_planner "${sid}"; return 0
  fi

  # ── D2 守卫: 检查 pane 是否被占用 ──
  local pane_idx="${pane##*.}"
  local current_sid="${PANE_CURRENT_SPRINT[$pane_idx]:-}"
  if [[ -n "$current_sid" ]] && [[ "$current_sid" != "$sid" ]]; then
    local assign_ts="${PANE_ASSIGN_TS[$pane_idx]:-0}"
    local elapsed=$(( $(date +%s) - assign_ts ))
    if (( elapsed < PANE_OCCUPY_TIMEOUT_SEC )); then
      log "${Y}[dispatch] pane ${pane} 仍归属 ${current_sid} (elapsed ${elapsed}s), 拒派 ${sid}${N}"
      emit_event "$sid" "dispatch_blocked" "coordinator" \
        "{\"pane\":\"${pane}\",\"reason\":\"pane_assigned\",\"current_sid\":\"${current_sid}\",\"elapsed_sec\":${elapsed}}"
      return 2   # 区别返回码: 2 = 阻塞 (调用方放回队列), 1 = 真失败
    fi
    # 超时 → 强制重派 (clear 旧 assignment 让下面正常 set)
    log "${Y}[dispatch] pane ${pane} 占用超时 (${elapsed}s ≥ ${PANE_OCCUPY_TIMEOUT_SEC}), 强制重派${N}"
    unset 'PANE_CURRENT_SPRINT[$pane_idx]'
    unset 'PANE_ASSIGN_TS[$pane_idx]'
  fi

  # ── 原有逻辑保持: is_pane_present / wait_for_dispatch_window / send-keys / verify ──
  # ...

  # ── D2 后置: 派发成功后 set assignment ──
  # 在 send-keys verify 双命中 return 0 之前 (line ~452):
  PANE_CURRENT_SPRINT[$pane_idx]="$sid"
  PANE_ASSIGN_TS[$pane_idx]=$(date +%s)
  save_pane_assignments    # 持久化, 见 D3
  log "${G}已派发到 ${pane}: ${instruction_file} ... [assign=${sid}]${N}"
  return 0
}
```

**自检**: `grep -nE 'return 2|dispatch_blocked|PANE_ASSIGN_TS\[.*\]=\$\(date'` ≥ 3.

rollback_state_cache() {
  local sid="$1"
  local current
  current=$(load_last_state)
  local kept=""
  while IFS='|' read -ra entries; do
    for e in "${entries[@]}"; do
      [[ -z "$e" ]] && continue
      [[ "${e%%:*}" == "$sid" ]] && continue   # 删除目标 sid

    done
  done <<< "$current"

### D3 — pane assignment 持久化

**位置**: `coordinator.sh` 顶部 (D1 之后).

```bash
# 持久化 assignment 到 .pane-assignments
# 格式: 每行 "idx=sid:ts", 例 "1=sprint-20260503-094659:1715000000"
save_pane_assignments() {
  local out=""
  for idx in "${!PANE_CURRENT_SPRINT[@]}"; do
    local sid="${PANE_CURRENT_SPRINT[$idx]}"
    local ts="${PANE_ASSIGN_TS[$idx]:-0}"
    [[ -z "$sid" ]] && continue
    out+="${idx}=${sid}:${ts}"$'\n'
  done
  # atomic write
  local tmp
  tmp=$(mktemp "$HARNESS_DIR/.pane-assignments.XXXXXX")
  printf '%s' "$out" > "$tmp"
  mv "$tmp" "$PANE_ASSIGNMENT_FILE"
}

# 启动时 reload
load_pane_assignments() {
  [[ -f "$PANE_ASSIGNMENT_FILE" ]] || return 0
  local count=0
  while IFS= read -r line; do
    [[ -z "$line" ]] && continue
    local idx="${line%%=*}"
    local rest="${line#*=}"
    local sid="${rest%:*}"
    local ts="${rest##*:}"
    PANE_CURRENT_SPRINT[$idx]="$sid"
    PANE_ASSIGN_TS[$idx]="$ts"
    ((count++))
  done < "$PANE_ASSIGNMENT_FILE"
  log "${G}loaded ${count} pane assignments from ${PANE_ASSIGNMENT_FILE}${N}"
}
```

**调用点**: `load_pane_assignments` 在主循环开始前调一次 (coord.sh 大约 line ~1820, startup recovery 段附近).

**自检**: `grep -cE 'save_pane_assignments|load_pane_assignments|\.pane-assignments'` ≥ 5.

### D4 — per-pane flock

**位置**: `dispatch_to_pane` 主体 (D2 守卫之后, send-keys 之前).

```bash
dispatch_to_pane() {
  # ... D2 守卫 ...

  # ── D4: per-pane flock ──
  local lock_file="$HARNESS_DIR/.dispatch-pane-${pane_idx}.lock"
  local lock_fd
  exec {lock_fd}>"$lock_file"
  if ! flock -n "$lock_fd"; then
    exec {lock_fd}>&-
    log "${Y}[dispatch] pane ${pane} flock 忙, 拒派 ${sid}${N}"
    emit_event "$sid" "dispatch_blocked" "coordinator" \
      "{\"pane\":\"${pane}\",\"reason\":\"flock_busy\"}"
    return 2
  fi

  # ── 原有 send-keys / verify 逻辑 ──
  # ...

  # 函数所有 return 出口前都释放
  flock -u "$lock_fd"
  exec {lock_fd}>&-
  return 0   # (或 1)
}
```

**注意**: bash 的 `exec {fd}>` 句法是 dynamic fd, 需要 bash 4.1+. 当前 coordinator.sh 已守卫 bash 4+ (line 49), 安全.

**陷阱**: 如果 dispatch_to_pane 中途 return 1 (失败), 必须释放锁. 用 trap 兜底:
```bash
trap "[[ -n \"\${lock_fd:-}\" ]] && exec \${lock_fd}>&- 2>/dev/null" RETURN
```

**自检**: `grep -cE 'flock -n|flock -u|\.dispatch-pane-'` ≥ 3.
