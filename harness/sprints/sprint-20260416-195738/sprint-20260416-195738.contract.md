```bash
handle_passed() {
  local sid="$1" sf="$2"
  # ... 原有通知派发逻辑 ...

  # 自动归档 (兜底不阻塞)
  bash "$HARNESS_DIR/archive.sh" auto 2>&1 | head -5 || true

}
```

### D2: generate_dispatch 推广

```
原: 每个 handle_* 用 cat > dispatch.md << EOF ... EOF
新: 统一调用 generate_dispatch "$sid" "$role" "$task_desc"

改动点:
  handle_active        → generate_dispatch $sid builder "写实现计划"
  handle_planning      → generate_dispatch $sid evaluator "审批计划"
  handle_approved      → generate_dispatch $sid builder "按计划实现"
  handle_reviewing     → generate_dispatch $sid evaluator "评审实现"
```

### D5: test-deadhead.sh 示例

```bash
#!/bin/bash
pass=0; fail=0

# T1: archive 有内容
if [ -n "$(ls ~/.solar/harness/sprints/archive/ 2>/dev/null)" ]; then
  echo "✅ T1"; pass=$((pass+1))
else
  echo "❌ T1: archive 空, D1 断头"; fail=$((fail+1))
fi

# T2: CACHE_BOUNDARY
latest=$(ls -t ~/.solar/harness/sprints/*.dispatch.md | head -1)
if grep -q "CACHE_BOUNDARY" "$latest"; then
  echo "✅ T2"; pass=$((pass+1))
else
  echo "❌ T2: 无 CACHE_BOUNDARY, D2 断头"; fail=$((fail+1))
fi

# T3: token 报告
count=$(ls ~/.solar/reports/token-savings-*.md 2>/dev/null | wc -l)
if [ "$count" -ge 2 ]; then
  echo "✅ T3 ($count 个)"; pass=$((pass+1))
else
  echo "❌ T3: 只 $count 个, D3 断头"; fail=$((fail+1))
fi

echo ""
echo "通过 $pass / 失败 $fail"
[ "$fail" -eq 0 ] && exit 0 || exit 1
```
