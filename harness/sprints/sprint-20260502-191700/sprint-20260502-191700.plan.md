### 1. `~/.solar/harness/lib/persona-config.sh` (新增)

共享配置函数，pane-launcher 和 start-incarnation 统一调用。

```bash
get_persona_config() {
  local persona="$1"
  # 从 model-config.sh 读 ZHIPU_* 变量 (不再硬编码)
  source "$HARNESS_DIR/model-config.sh" 2>/dev/null || true

  case "$persona" in
    planner)
      echo "MODEL_FLAG=--model opus"
      echo "BASE_URL="           # unset → 直连 Anthropic
      echo "AUTH_TOKEN="         # unset
      echo "TOOL_FLAG=--allowedTools Read Bash Grep Glob"
      echo "DISPLAY_MODEL=Claude Opus (Anthropic)"
      echo "STARTUP_TOKEN=solar"
      echo "PROXY_CHECK=1"
      ;;
    builder)
      echo "MODEL_FLAG=--model $ZHIPU_MODEL"
      echo "BASE_URL=$ZHIPU_BASE_URL"
      echo "AUTH_TOKEN=$ZHIPU_AUTH_TOKEN"
      echo "TOOL_FLAG="
      echo "DISPLAY_MODEL=GLM-5.1 (智谱)"
      echo "STARTUP_TOKEN="      # 只按 Enter
      echo "PROXY_CHECK=0"
      ;;
    evaluator)
      echo "MODEL_FLAG=--model $ZHIPU_MODEL"
      echo "BASE_URL=$ZHIPU_BASE_URL"
      echo "AUTH_TOKEN=$ZHIPU_AUTH_TOKEN"
      echo "TOOL_FLAG=--allowedTools Read Bash Grep Glob Write"
      echo "DISPLAY_MODEL=GLM-5.1 (智谱)"
      echo "STARTUP_TOKEN="      # 只按 Enter
      echo "PROXY_CHECK=0"
      ;;
  esac
}
```

同时提供 `--print-config <persona>` CLI 接口，供 D1 verify 命令使用。
